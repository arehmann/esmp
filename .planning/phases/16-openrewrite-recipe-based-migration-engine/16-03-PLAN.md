    ```java
    @RestController
    @RequestMapping("/api/migration")
    public class MigrationController {

        private final MigrationRecipeService migrationRecipeService;

        public MigrationController(MigrationRecipeService migrationRecipeService) {
            this.migrationRecipeService = migrationRecipeService;
        }

        // GET /api/migration/plan/{fqn:.+}
        // Returns MigrationPlan for the class
        // 404 if class not found (generatePlan returns null or throws)
        @GetMapping("/plan/{fqn:.+}")
        public ResponseEntity<MigrationPlan> getPlan(@PathVariable String fqn) {
            MigrationPlan plan = migrationRecipeService.generatePlan(fqn);
            if (plan == null) return ResponseEntity.notFound().build();
            return ResponseEntity.ok(plan);
        }

        // GET /api/migration/summary?module=xxx
        // Returns ModuleMigrationSummary
        @GetMapping("/summary")
        public ResponseEntity<ModuleMigrationSummary> getSummary(@RequestParam String module) {
            if (module == null || module.isBlank()) return ResponseEntity.badRequest().build();
            return ResponseEntity.ok(migrationRecipeService.getModuleSummary(module));
        }

        // POST /api/migration/preview/{fqn:.+}
        // Returns MigrationResult with diff (no file write)
        @PostMapping("/preview/{fqn:.+}")
        public ResponseEntity<MigrationResult> preview(@PathVariable String fqn) {
            MigrationResult result = migrationRecipeService.preview(fqn);
            return ResponseEntity.ok(result);
        }

        // POST /api/migration/apply/{fqn:.+}
        // Applies recipes and writes to disk
        @PostMapping("/apply/{fqn:.+}")
        public ResponseEntity<MigrationResult> apply(@PathVariable String fqn) {
            MigrationResult result = migrationRecipeService.applyAndWrite(fqn);
            return ResponseEntity.ok(result);
        }

        // POST /api/migration/apply-module
        // Batch applies all automatable recipes in a module
        @PostMapping("/apply-module")
        public ResponseEntity<BatchMigrationResult> applyModule(@RequestBody ModuleMigrationRequest req) {
            if (req.module() == null || req.module().isBlank()) return ResponseEntity.badRequest().build();
            BatchMigrationResult result = migrationRecipeService.applyModule(req.module());
            return ResponseEntity.ok(result);
        }
    }
    ```
    - Use `{fqn:.+}` regex suffix on path variables to prevent Spring MVC dot-truncation (same pattern as GraphQueryController)
    - Add error handling: wrap service calls in try/catch, return 500 with error message body for unexpected failures, 404 for class-not-found
