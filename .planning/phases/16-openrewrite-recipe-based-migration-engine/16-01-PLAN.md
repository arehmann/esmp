    List<MigrationActionData> actions = acc.getMigrationActions().getOrDefault(fqn, List.of());
    int totalActions = actions.size();
    int yesCount = (int) actions.stream().filter(a -> a.automatable() == Automatable.YES).count();
    int partialCount = (int) actions.stream().filter(a -> a.automatable() == Automatable.PARTIAL).count();
    boolean hasNo = actions.stream().anyMatch(a -> a.automatable() == Automatable.NO);
    classNode.setMigrationActionCount(totalActions);
    classNode.setAutomatableActionCount(yesCount);
    classNode.setAutomationScore(totalActions > 0 ? (yesCount + 0.5 * partialCount) / totalActions : 0.0);
