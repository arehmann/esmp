**6. Create `RecipeBookController`** at `src/main/java/com/esmp/migration/api/RecipeBookController.java`:
```java
@RestController
@RequestMapping("/api/migration/recipe-book")
public class RecipeBookController {
    private final RecipeBookRegistry recipeBookRegistry;

    // GET /api/migration/recipe-book?category=&status=&automatable=
    @GetMapping
    public List<RecipeRule> getAllRules(
        @RequestParam(required = false) String category,
        @RequestParam(required = false) String status,
        @RequestParam(required = false) String automatable) {
        return recipeBookRegistry.getRules().stream()
            .filter(r -> category == null || category.equals(r.category()))
            .filter(r -> status == null || status.equals(r.status()))
            .filter(r -> automatable == null || automatable.equals(r.automatable()))
            .collect(Collectors.toList());
    }

    // GET /api/migration/recipe-book/gaps
    @GetMapping("/gaps")
    public List<RecipeRule> getGaps() {
        return recipeBookRegistry.getRules().stream()
            .filter(r -> "NEEDS_MAPPING".equals(r.status()))
            .sorted(Comparator.comparingInt(RecipeRule::usageCount).reversed())
            .collect(Collectors.toList());
    }

    // PUT /api/migration/recipe-book/rules/{id}
    @PutMapping("/rules/{id}")
    public ResponseEntity<RecipeRule> upsertRule(@PathVariable String id, @RequestBody RecipeRule rule) {
        // Create rule with provided id, isBase=false
        RecipeRule newRule = new RecipeRule(id, rule.category(), rule.source(), rule.target(),
            rule.actionType(), rule.automatable(), rule.context(), rule.migrationSteps(),
            rule.status(), rule.usageCount(), rule.discoveredAt(), false);
        List<RecipeRule> updated = new ArrayList<>(recipeBookRegistry.getRules());
        // Replace existing rule with same id, or add new
        updated.removeIf(r -> r.id().equals(id));
        updated.add(newRule);
        recipeBookRegistry.updateAndWrite(updated);
        return ResponseEntity.ok(newRule);
    }

    // DELETE /api/migration/recipe-book/rules/{id}
    @DeleteMapping("/rules/{id}")
    public ResponseEntity<Void> deleteRule(@PathVariable String id) {
        Optional<RecipeRule> existing = recipeBookRegistry.getRules().stream()
            .filter(r -> r.id().equals(id)).findFirst();
        if (existing.isEmpty()) return ResponseEntity.notFound().build();
        if (existing.get().isBase()) return ResponseEntity.status(403).build(); // base rules protected
        List<RecipeRule> updated = recipeBookRegistry.getRules().stream()
            .filter(r -> !r.id().equals(id))
            .collect(Collectors.toList());
        recipeBookRegistry.updateAndWrite(updated);
        return ResponseEntity.noContent().build();
    }

    // POST /api/migration/recipe-book/reload
    @PostMapping("/reload")
    public ResponseEntity<Void> reload() {
        recipeBookRegistry.reload();
        return ResponseEntity.ok().build();
    }
}
```
